# Keep only Healthy Control and ALS
metadata_sub <- metadata %>%

## [1] 49 10
# sample names in metadata are in the title column
metadata_sub$sample <- metadata_sub$title

# keep only count columns for those samples
keep_samples <- metadata_sub$sample
counts_sub <- counts[, c("gene_id", keep_samples), with = FALSE]

# make count matrix
count_matrix <- as.data.frame(counts_sub)
rownames(count_matrix) <- count_matrix$gene_id
count_matrix$gene_id <- NULL
count_matrix <- as.matrix(count_matrix)

# reorder metadata to match count matrix
metadata_sub <- metadata_sub[match(colnames(count_matrix), metadata_sub$sample), ]

# check
all(colnames(count_matrix) == metadata_sub$sample)
## [1] TRUE
# set factor levels
metadata_sub$diseasestatus_ch1 <- factor(
  metadata_sub$diseasestatus_ch1,
  levels = c("Healthy Control", "ALS")
)

table(metadata_sub$diseasestatus_ch1)

## Healthy Control             ALS 
##              28              21
# create DESeq2 dataset
dds <- DESeqDataSetFromMatrix(
  countData = count_matrix,
  colData = metadata_sub,
  design = ~ diseasestatus_ch1
)
